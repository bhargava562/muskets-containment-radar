## рҹ“Ҡ TECHNICAL IMPLEMENTATION

### PII Masking Logic
```javascript
const maskPII = (identifier) => {
  // Account numbers (10+ digits)
  if (/^\d{10,}$/.test(identifier)) {
    return `${prefix}*****${suffix}` // Keep first 4 and last 4
  }
  
  // IFSC codes (AAAA0000000)
  if (/^[A-Z]{4}\d{7}$/.test(identifier)) {
    return `${prefix}*****${suffix}`
  }
  
  // Node IDs (TYPE_ID)
  if (identifier.includes('_')) {
    return `${type}_****`
  }
  
  // Generic fallback
  return `${start}*****${end}`
}
```
